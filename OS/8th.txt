8. Implement page replacement algorithm LRU with suitable page request and display the number of page faults.                                        

#include <stdio.h>
#include <stdlib.h>

void display(int fr[], int frsize) {
    for (int i = 0; i < frsize; i++) {
        printf("\t%d", fr[i]);
    }
    printf("\n");
}

int main() {
    int pages[12] = {2, 3, 2, 1, 5, 2, 4, 5, 3, 2, 5, 2};
    int n = 12;
    int frsize = 3;
    int fr[3], fs[3];
    int i, j, k, l, index = 0;
    int pf = 0;
    int flag1, flag2;

    // Initialize frame with -1
    for (i = 0; i < frsize; i++) {
        fr[i] = -1;
    }

    for (j = 0; j < n; j++) {
        flag1 = flag2 = 0;

        // Check if page is already in frame
        for (i = 0; i < frsize; i++) {
            if (fr[i] == pages[j]) {
                flag1 = flag2 = 1;
                break;
            }
        }

        // If page is not in frame, check for empty slot
        if (flag1 == 0) {
            for (i = 0; i < frsize; i++) {
                if (fr[i] == -1) {
                    fr[i] = pages[j];
                    flag2 = 1;
                    pf++;
                    break;
                }
            }
        }

        // If no empty slot and not found, do LRU replacement
        if (flag2 == 0) {
            for (i = 0; i < frsize; i++) {
                fs[i] = 0;
            }

            for (k = j - 1, l = 1; l <= frsize - 1 && k >= 0; k--) {
                for (i = 0; i < frsize; i++) {
                    if (fr[i] == pages[k] && fs[i] == 0) {
                        fs[i] = 1;
                        l++;
                        break;
                    }
                }
            }

            for (i = 0; i < frsize; i++) {
                if (fs[i] == 0) {
                    index = i;
                    break;
                }
            }

            fr[index] = pages[j];
            pf++;
        }

        display(fr, frsize);
    }

    printf("\nTotal number of page faults: %d\n", pf);
    return 0;
}
